    #reorder to have 'file_name' as first column
    
    selectfloat = selectfloat[['file_name', 'thickness']]
    
    
    # Append the filtered values to the combined_results list
    if not selectfloat.empty:
        combined_results.append(selectfloat)

    if not selectfloat.empty:
